# By Animesh Sitoula | Python + JSON

# ok so first we import what we need
# json for saving/loading data, os to check if files exist
import json as j
import os

# these are the subjects - stored as a tuple so no one messes with them
subjects = ("math", "physics", "chemistry", "computer", "english", "nepali")

# tracking who got the highest marks in each subject
subject_maximum_marks = {subj: 0 for subj in subjects}
subject_topper        = {subj: "N/A" for subj in subjects}

def add_student_record(no_of_students):
    # loop through each student and collect their info
    student_record = []
    for i in range(1, no_of_students + 1):
        total_marks = 0
        record = {
            "name": str(input(f"Enter name of student {i}: ")),
            "age":  int(input(f"Enter age of student {i}: "))
        }
        for subj in subjects:
            marks = int(input(f"Enter marks in {subj}: "))
            # validate marks - max is 75 per subject
            if marks > 75:
                raise ValueError("Marks must not exceed 75")
            record[f"marks in {subj}"] = marks
            total_marks += marks
        record["total_marks"]   = total_marks
        record["average_marks"] = round(total_marks / len(subjects), 2)
        # pass only if EVERY subject has at least 28 marks
        record["Result"] = "Pass" if all(
            record[f"marks in {s}"] >= 28 for s in subjects
        ) else "Fail"
        student_record.append(record)

    # now save to JSON - append to existing file if it exists
    filename = "student_record.json"
    data = []
    if os.path.exists(filename):
        with open(filename, "r") as f:
            try:    data = j.load(f)
            except: data = []  # file might be empty/corrupted
    data.extend(student_record)
    with open(filename, "w") as f:
        j.dump(data, f, indent=4)
    print(f"[+] {no_of_students} record(s) saved.")

def view_student_record():
    # show all stored records - quit early if no file found
    if not os.path.exists("student_record.json"):
        print("No records found."); return
    with open("student_record.json", "r") as f:
        data = j.load(f)
    for student in data:
        print("-" * 35)
        for key, val in student.items():
            print(f"  {key}: {val}")

# ===== MAIN MENU LOOP =====
while True:
    print("\n====== STUDENT PERFORMANCE ANALYTICS ======")
    print("1. Add student records")
    print("2. View all student reports")
    print("5. Exit")
    try:
        choice = int(input("Enter choice: "))
    except ValueError:
        print("Enter a valid number."); continue
    if choice == 1:
        add_student_record(int(input("Number of students: ")))
    elif choice == 2: view_student_record()
    elif choice == 5: print("Bye!"); break

# Main App class - everything lives inside this
class StudentApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Student Manager")
        self.root.geometry("600x500")
        self.root.configure(bg="#1e1e2e")
        self.build_ui()

        # Header label at the top
        header = tk.Label(
            self.root, text="STUDENT MANAGER",
            font=("Courier", 16, "bold"),
            bg="#1e1e2e", fg="#00ff88"
        )
        header.pack(pady=20)

        # Input frame using grid layout
        frame = tk.Frame(self.root, bg="#1e1e2e")
        frame.pack(fill="x", padx=40)

        # Name field
        tk.Label(frame, text="Name:", bg="#1e1e2e", fg="#c8d4e0",
                 font=("Courier", 10)).grid(row=0, column=0, sticky="w", pady=5)
        self.name_var = tk.StringVar()
        self.name_entry = tk.Entry(frame, textvariable=self.name_var,
                                      bg="#0c1520", fg="white",
                                      font=("Courier", 10), width=30,
                                      insertbackground="#00ff88")
        self.name_entry.grid(row=0, column=1, padx=10)

        # Submit button with event binding
        btn = tk.Button(
            self.root, text="ADD STUDENT",
            command=self.add_student,
            bg="#00ff88", fg="#060a0e",
            font=("Courier", 10, "bold"),
            relief="flat", padx=20, pady=8
        )
        btn.pack(pady=20)
        # bind Enter key as well - more user friendly
        self.root.bind("<Return>", lambda e: self.add_student())

    def add_student(self):
        name = self.name_var.get().strip()
        if not name:
            # show error popup if field is empty
            messagebox.showerror("Error", "Name cannot be empty!")
            return
        messagebox.showinfo("Success", f"Added: {name}")
        self.name_var.set("")  # clear the field after adding

# Start the app - this is how every tkinter app launches
root = tk.Tk()
app = StudentApp(root)
root.mainloop()  # keeps the window alive

# Network Mapper — multithreaded subnet scanner
# By Animesh Sitoula | Python + Sockets + Threading
# =================================================
import ipaddress
import socket
import subprocess
from concurrent.futures import ThreadPoolExecutor

# ping one host and return True if it's alive
def ping_host(ip):
    # subprocess runs actual system ping command
    result = subprocess.run(
        ["ping", "-c", "1", "-W", "1", str(ip)],
        capture_output=True
    )
    return result.returncode == 0

# try to get hostname from IP - reverse DNS lookup
def get_hostname(ip):
    try:
        return socket.gethostbyaddr(str(ip))[0]
    except:
        return "Unknown"  # if reverse DNS fails, just say Unknown

# scan one IP - ping it then resolve hostname if alive
def scan_host(ip):
    if ping_host(ip):
        hostname = get_hostname(ip)
        print(f"  [+] ALIVE  {str(ip):<18} {hostname}")
        return (ip, hostname)
    return None

# main function - takes a subnet like "192.168.1.0/24"
def map_network(subnet, max_threads=50):
    network = ipaddress.ip_network(subnet, strict=False)
    hosts = list(network.hosts())
    print(f"\n[*] Scanning {len(hosts)} hosts on {subnet}...\n")

    active = []
    # ThreadPoolExecutor runs multiple scans at the same time - way faster!
    with ThreadPoolExecutor(max_workers=max_threads) as pool:
        results = pool.map(scan_host, hosts)
    active = [r for r in results if r is not None]

    print(f"\n[+] Done. Found {len(active)} active hosts.")

# run it
if __name__ == "__main__":
    target = input("Enter subnet (e.g. 192.168.1.0/24): ")
    map_network(target)

# TCP Port Scanner with banner grabbing
# By Animesh Sitoula | Python + Sockets + Threading
# =================================================
import socket
import threading
from datetime import datetime

# map of common ports to service names
COMMON_PORTS = {
    21: "FTP",  22: "SSH", 23: "Telnet",
    25: "SMTP", 53: "DNS", 80: "HTTP",
    443: "HTTPS", 3306: "MySQL", 8080: "HTTP-Alt"
}

open_ports = []
lock = threading.Lock()  # prevents thread conflicts when writing to list

def scan_port(target, port):
    # try connecting to the port - if successful it's open
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(1)  # don't wait too long
        result = sock.connect_ex((target, port))
        if result == 0:
            # port is open! try to get service banner
            banner = ""
            try:
                sock.send(b"HEAD / HTTP/1.0\r\n\r\n")
                banner = sock.recv(1024).decode("utf-8", errors="ignore").strip()[:50]
            except: pass
            service = COMMON_PORTS.get(port, "Unknown")
            # thread-safe append
            with lock:
                open_ports.append((port, service, banner))
        sock.close()
    except: pass

# scan a range of ports using threads
def run_scan(target, start=1, end=1024):
    print(f"\n[*] Scanning {target} ports {start}-{end}")
    t0 = datetime.now()
    threads = []
    for port in range(start, end + 1):
        t = threading.Thread(target=scan_port, args=(target, port))
        threads.append(t)
        t.start()
    for t in threads:
        t.join()  # wait for all threads to finish
    elapsed = (datetime.now() - t0).seconds
    print(f"\n[+] Scan done in {elapsed}s. {len(open_ports)} open port(s).")
    for port, svc, banner in sorted(open_ports):
        print(f"  PORT {port:<6} {svc:<12} {banner}")

if __name__ == "__main__":
    host = input("Target IP/host: ")
    run_scan(host)

# TCP Server — handles multiple clients with threads
# By Animesh Sitoula
# =================================================
import socket
import threading

HOST = "0.0.0.0"  # listen on all interfaces
PORT = 9999

# this runs in a separate thread for each client that connects
def handle_client(conn, addr):
    print(f"[+] New connection from {addr}")
    try:
        while True:
            data = conn.recv(1024)
            if not data:
                break  # client disconnected
            msg = data.decode("utf-8")
            print(f"  [{addr[0]}] {msg}")
            # echo back to client with a prefix
            conn.send(f"SERVER: {msg}".encode())
    except:
        print(f"[-] {addr} disconnected")
    finally:
        conn.close()

def start_server():
    # create socket - AF_INET means IPv4, SOCK_STREAM means TCP
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    server.bind((HOST, PORT))
    server.listen(5)  # accept up to 5 queued connections
    print(f"[*] Server listening on {HOST}:{PORT}")

    while True:
        conn, addr = server.accept()
        # spawn a new thread for each client so they dont block each other
        t = threading.Thread(target=handle_client, args=(conn, addr))
        t.daemon = True
        t.start()

start_server()

# TCP Client — connects to the server and chats
# By Animesh Sitoula
# =================================================
import socket
import threading

HOST = "127.0.0.1"  # connect to localhost for testing
PORT = 9999

# this thread listens for messages from server in background
def receive_msgs(sock):
    while True:
        try:
            data = sock.recv(1024)
            if not data: break
            print(f"\r{data.decode('utf-8')}\n> ", end="")
        except: break

def start_client():
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.connect((HOST, PORT))
    print(f"[+] Connected to {HOST}:{PORT}")

    # start listener thread in background - daemon means it dies with main thread
    t = threading.Thread(target=receive_msgs, args=(sock,), daemon=True)
    t.start()

    # main loop - type messages and send them
    while True:
        msg = input("> ")
        if msg.lower() in ("quit", "exit"):
            break
        sock.send(msg.encode("utf-8"))

    sock.close()

start_client()


  
/* =================================================
   Caesar Cipher in C - encrypt and decrypt text
   By Animesh Sitoula
   ================================================= */
#include <stdio.h>
#include <string.h>

#define SHIFT 3  /* the Caesar shift - change this to use different key */
#define MAX   256

/* encrypt - shifts each letter forward by SHIFT positions */
void encrypt(char *text) {
    for (int i = 0; text[i] != '\0'; i++) {
        if (text[i] >= 'A' && text[i] <= 'Z')
            text[i] = (char)(((text[i] - 'A' + SHIFT) % 26) + 'A');
        else if (text[i] >= 'a' && text[i] <= 'z')
            text[i] = (char)(((text[i] - 'a' + SHIFT) % 26) + 'a');
        /* non-letters stay the same */
    }
}

/* decrypt - shifts backward by SHIFT (same as encrypting with 26-SHIFT) */
void decrypt(char *text) {
    for (int i = 0; text[i] != '\0'; i++) {
        if (text[i] >= 'A' && text[i] <= 'Z')
            text[i] = (char)(((text[i] - 'A' - SHIFT + 26) % 26) + 'A');
        else if (text[i] >= 'a' && text[i] <= 'z')
            text[i] = (char)(((text[i] - 'a' - SHIFT + 26) % 26) + 'a');
    }
}

int main() {
    char text[MAX];
    int choice;

    printf("1. Encrypt\n2. Decrypt\nChoice: ");
    scanf("%d", &choice);
    getchar(); /* consume leftover newline */

    printf("Enter text: ");
    fgets(text, MAX, stdin);

    /* fgets includes \n at end - strip it */
    text[strcspn(text, "\n")] = '\0';

    if (choice == 1) {
        encrypt(text);
        printf("Encrypted: %s\n", text);
    } else {
        decrypt(text);
        printf("Decrypted: %s\n", text);
    }
    return 0;
}
