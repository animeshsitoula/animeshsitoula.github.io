# Student Performance Analysis System
# Technology: Python, JSON — by Animesh Sitoula

import json as j
import os

subjects = ("math", "physics", "chemistry", "computer", "english", "nepali")
subject_maximum_marks = {subj: 0 for subj in subjects}
subject_topper        = {subj: "N/A" for subj in subjects}

def add_student_record(no_of_students):
    student_record = []
    for i in range(1, no_of_students + 1):
        total_marks = 0
        record = {
            "name": str(input(f"Enter name of student {i}: ")),
            "age":  int(input(f"Enter age of student {i}: "))
        }
        for subj in subjects:
            marks = int(input(f"Enter marks in {subj}: "))
            if marks > 75:
                raise ValueError("Marks must not exceed 75")
            record[f"marks in {subj}"] = marks
            total_marks += marks
        record["total_marks"]   = total_marks
        record["average_marks"] = round(total_marks / len(subjects), 2)
        record["Result"] = "Pass" if all(
            record[f"marks in {s}"] >= 28 for s in subjects
        ) else "Fail"
        student_record.append(record)

    filename = "student_record.json"
    data = []
    if os.path.exists(filename):
        with open(filename, "r") as f:
            try:    data = j.load(f)
            except: data = []
    data.extend(student_record)
    with open(filename, "w") as f:
        j.dump(data, f, indent=4)
    print(f"[+] {no_of_students} record(s) saved.")

def view_student_record():
    if not os.path.exists("student_record.json"):
        print("No records found."); return
    with open("student_record.json", "r") as f:
        data = j.load(f)
    for student in data:
        print("-" * 35)
        for key, val in student.items():
            print(f"  {key}: {val}")

def subj_wise_topper():
    if not os.path.exists("student_record.json"):
        print("No records found."); return
    with open("student_record.json", "r") as f:
        data = j.load(f)
    for student in data:
        for subject in subjects:
            if student[f"marks in {subject}"] > subject_maximum_marks[subject]:
                subject_maximum_marks[subject] = student[f"marks in {subject}"]
                subject_topper[subject]        = student["name"]
    print("\nSubject-wise Toppers:")
    for subj in subject_topper:
        print(f"  {subj.title()}: {subject_topper[subj]} ({subject_maximum_marks[subj]} marks)")

def class_rank():
    if not os.path.exists("student_record.json"):
        print("No records found."); return
    with open("student_record.json", "r") as f:
        data = j.load(f)
    ranked = sorted(data, key=lambda x: x["total_marks"], reverse=True)
    print("\nClass Ranking:")
    for i, s in enumerate(ranked, 1):
        print(f"  {i}. {s['name']} — {s['total_marks']} marks  [{s['Result']}]")

# ===== MAIN MENU =====
while True:
    print("\n====== STUDENT PERFORMANCE ANALYTICS ======")
    print("1. Add student records")
    print("2. View all student reports")
    print("3. View subject-wise toppers")
    print("4. View class rank list")
    print("5. Exit")
    try:
        choice = int(input("Enter choice: "))
    except ValueError:
        print("Enter a valid number."); continue
    if   choice == 1:
        try: add_student_record(int(input("Number of students: ")))
        except ValueError: print("Invalid input.")
    elif choice == 2: view_student_record()
    elif choice == 3: subj_wise_topper()
    elif choice == 4: class_rank()
    elif choice == 5: print("Goodbye!"); break
    else: print("Invalid choice.")

import tkinter as tk
from tkinter import messagebox

class StudentApp(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Student Info Manager")
        self.geometry("520x400")
        self.configure(bg="#1e1e2e")
        self.resizable(False, False)
        self.records = []
        self._build_ui()

    def _build_ui(self):
        hdr = tk.Frame(self, bg="#0d1117", pady=10)
        hdr.pack(fill="x")
        tk.Label(hdr, text="STUDENT MANAGER",
            font=("Courier", 13, "bold"),
            bg="#0d1117", fg="#00ff88").pack()

        form = tk.Frame(self, bg="#1e1e2e", padx=30, pady=15)
        form.pack(fill="both", expand=True)

        fields = [("Name", "name"), ("Age", "age"),
                  ("Math Score", "math"), ("English Score", "english")]
        self.entries = {}

        for row, (label, key) in enumerate(fields):
            tk.Label(form, text=f"{label}:",
                font=("Courier", 9), bg="#1e1e2e", fg="#c9d1d9",
                anchor="w"
            ).grid(row=row, column=0, sticky="w", pady=4)
            entry = tk.Entry(form, width=28, font=("Courier", 9),
                bg="#0d1520", fg="#c9d1d9",
                insertbackground="#00ff88", relief="flat")
            entry.grid(row=row, column=1, padx=(10,0), pady=4)
            self.entries[key] = entry

        btn_frame = tk.Frame(self, bg="#1e1e2e", pady=10)
        btn_frame.pack()
        tk.Button(btn_frame, text="ADD RECORD", command=self.add_record,
            bg="#00ff88", fg="#000", font=("Courier",9,"bold"),
            relief="flat", padx=12, pady=6
        ).pack(side="left", padx=5)
        tk.Button(btn_frame, text="VIEW ALL", command=self.view_records,
            bg="#0d1520", fg="#00d4ff", font=("Courier",9),
            relief="flat", padx=12, pady=6
        ).pack(side="left", padx=5)

        self.status = tk.StringVar(value="Ready.")
        tk.Label(self, textvariable=self.status,
            font=("Courier",8), bg="#0d1117",
            fg="#556070", anchor="w", padx=10
        ).pack(fill="x", side="bottom")

    def add_record(self):
        vals = {k: e.get().strip() for k, e in self.entries.items()}
        if not all(vals.values()):
            messagebox.showwarning("Missing", "Fill all fields."); return
        try:
            record = {"name": vals["name"], "age": int(vals["age"]),
                      "math": int(vals["math"]), "english": int(vals["english"])}
        except ValueError:
            messagebox.showerror("Error", "Age and scores must be numbers."); return
        self.records.append(record)
        self.status.set(f"Record added — {len(self.records)} total.")
        for e in self.entries.values(): e.delete(0, "end")

    def view_records(self):
        if not self.records:
            messagebox.showinfo("Empty", "No records yet."); return
        win = tk.Toplevel(self)
        win.title("All Records"); win.configure(bg="#0d1117")
        txt = tk.Text(win, font=("Courier",9), bg="#0d1520",
                       fg="#c9d1d9", padx=10, pady=10)
        txt.pack(fill="both", expand=True, padx=10, pady=10)
        for r in self.records: txt.insert("end", f"  {r}\n")
        txt.config(state="disabled")

if __name__ == "__main__":
    StudentApp().mainloop()


  
# Network Mapper — Multithreaded Host Discovery
# by Animesh Sitoula

import socket, subprocess, ipaddress, platform
from concurrent.futures import ThreadPoolExecutor, as_completed
from datetime import datetime

def ping_host(ip):
    flag   = "-n" if platform.system() == "Windows" else "-c"
    result = subprocess.run(
        ["ping", flag, "1", "-W", "1", str(ip)],
        stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
    return ip if result.returncode == 0 else None

def resolve_host(ip):
    try:
        hostname, _, _ = socket.gethostbyaddr(str(ip))
        return hostname
    except socket.herror:
        return "Unknown"

def get_mac(ip):
    try:
        out = subprocess.check_output(
            ["arp", "-n", str(ip)], stderr=subprocess.DEVNULL).decode()
        for line in out.splitlines()[1:]:
            parts = line.split()
            if len(parts) >= 3: return parts[2]
    except Exception: pass
    return "N/A"

def scan_network(cidr, workers=64):
    print(f"\n[*] Scanning : {cidr}")
    print(f"[*] Started  : {datetime.now().strftime('%H:%M:%S')}\n")
    network = ipaddress.ip_network(cidr, strict=False)
    hosts   = list(network.hosts())
    active  = []
    with ThreadPoolExecutor(max_workers=workers) as ex:
        futures = {ex.submit(ping_host, ip): ip for ip in hosts}
        for fut in as_completed(futures):
            ip = fut.result()
            if ip:
                h   = resolve_host(ip)
                mac = get_mac(ip)
                active.append({"ip": str(ip), "hostname": h, "mac": mac})
                print(f"  [+] {str(ip):<18} {h:<30} {mac}")
    print(f"\n[*] Done — {len(active)} active device(s).")
    return active

if __name__ == "__main__":
    target = input("Enter CIDR (e.g. 192.168.1.0/24): ").strip()
    scan_network(target)


  
# Port Scanner — TCP Service Detection
# by Animesh Sitoula

import socket, sys, json
from concurrent.futures import ThreadPoolExecutor, as_completed
from datetime import datetime

COMMON_PORTS = {
    21:"FTP", 22:"SSH", 23:"Telnet", 25:"SMTP", 53:"DNS",
    80:"HTTP", 110:"POP3", 143:"IMAP", 443:"HTTPS",
    3306:"MySQL", 3389:"RDP", 8080:"HTTP-Alt"
}

def grab_banner(ip, port):
    try:
        s = socket.socket()
        s.settimeout(1.5)
        s.connect((ip, port))
        banner = s.recv(1024).decode(errors="ignore").strip()
        s.close()
        return banner[:60] if banner else "—"
    except: return "—"

def scan_port(ip, port):
    try:
        s  = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        s.settimeout(0.6)
        ok = s.connect_ex((ip, port)) == 0
        s.close()
        if ok:
            svc    = COMMON_PORTS.get(port, "Unknown")
            banner = grab_banner(ip, port)
            return {"port": port, "service": svc, "banner": banner}
    except: pass
    return None

def run_scan(target, port_range, workers=150):
    try: ip = socket.gethostbyname(target)
    except: print("Cannot resolve target."); sys.exit()
    print(f"\n[*] Target  : {target} ({ip})")
    print(f"[*] Started : {datetime.now().strftime('%H:%M:%S')}\n")
    open_ports = []
    with ThreadPoolExecutor(workers) as ex:
        futures = {ex.submit(scan_port, ip, p): p for p in port_range}
        for f in as_completed(futures):
            res = f.result()
            if res:
                open_ports.append(res)
                print(f"  [OPEN] {res['port']:<6} {res['service']:<14} {res['banner']}")
    open_ports.sort(key=lambda x: x["port"])
    print(f"\n[*] {len(open_ports)} open port(s) found.")
    with open("scan_results.json", "w") as f:
        json.dump({"target": ip, "ports": open_ports}, f, indent=2)

if __name__ == "__main__":
    run_scan(input("Target host: ").strip(), range(1, 1025))


  
# TCP Server — Client-Server Architecture
# by Animesh Sitoula

import socket, threading

HOST = "0.0.0.0"
PORT = 9090
clients      = {}
clients_lock = threading.Lock()

def broadcast(msg, exclude=None):
    with clients_lock:
        for conn in clients:
            if conn != exclude:
                try: conn.sendall(msg.encode())
                except: pass

def handle_client(conn, addr):
    print(f"[+] {addr} connected")
    try:
        conn.sendall(b"Enter your name: ")
        name = conn.recv(64).decode().strip() or f"User_{addr[1]}"
        with clients_lock: clients[conn] = name
        broadcast(f"[SERVER] {name} joined.\n", conn)
        conn.sendall(f"[SERVER] Welcome {name}! Type quit to exit.\n".encode())
        while True:
            data = conn.recv(1024)
            if not data: break
            msg = data.decode().strip()
            if msg.lower() == "quit": break
            print(f"[{name}] {msg}")
            broadcast(f"[{name}] {msg}\n", conn)
    except ConnectionResetError: pass
    finally:
        with clients_lock: name = clients.pop(conn, "Unknown")
        conn.close()
        broadcast(f"[SERVER] {name} left.\n")
        print(f"[-] {name} disconnected")

def start_server():
    srv = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    srv.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    srv.bind((HOST, PORT)); srv.listen(10)
    print(f"[*] Listening on {HOST}:{PORT}")
    while True:
        conn, addr = srv.accept()
        threading.Thread(target=handle_client,
            args=(conn, addr), daemon=True).start()

if __name__ == "__main__": start_server()


  
# TCP Client — Client-Server Architecture
# by Animesh Sitoula

import socket, threading, sys

HOST = "127.0.0.1"
PORT = 9090

def receive_messages(sock):
    while True:
        try:
            data = sock.recv(1024)
            if not data:
                print("\n[!] Server closed connection."); sys.exit()
            print(data.decode(), end="", flush=True)
        except ConnectionResetError:
            print("\n[!] Connection reset."); sys.exit()
        except Exception as e:
            print(f"\n[!] Error: {e}"); break

def start_client():
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    try:
        sock.connect((HOST, PORT))
        print(f"[*] Connected to {HOST}:{PORT}")
    except ConnectionRefusedError:
        print("[!] Connection refused. Is server running?"); sys.exit()

    threading.Thread(target=receive_messages,
        args=(sock,), daemon=True).start()
    try:
        while True:
            msg = input("")
            if not msg: continue
            sock.sendall(msg.encode())
            if msg.lower() == "quit": break
    except (KeyboardInterrupt, EOFError): print("\n[*] Disconnected.")
    finally: sock.close()

if __name__ == "__main__": start_client()


  
/* Cipher Encryption — C Language */
/* by Animesh Sitoula */

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#define MAX   256
#define ALPHA 26

void caesar(const char *in, char *out, int shift) {
    shift = ((shift % ALPHA) + ALPHA) % ALPHA;
    int n = strlen(in);
    for (int i = 0; i < n; i++) {
        char c = in[i];
        if      (isupper(c)) out[i] = (c - 'A' + shift) % ALPHA + 'A';
        else if (islower(c)) out[i] = (c - 'a' + shift) % ALPHA + 'a';
        else                  out[i] = c;
    }
    out[n] = '\0';
}

void print_hex(const char *s) {
    printf("Hex: ");
    for (int i = 0; s[i]; i++)
        printf("%02x ", (unsigned char)s[i]);
    printf("\n");
}

int main() {
    char plain[MAX], enc[MAX], dec[MAX];
    int  shift;
    printf("Plaintext : ");
    fgets(plain, MAX, stdin);
    plain[strcspn(plain, "\n")] = 0;
    printf("Shift key : ");
    scanf("%d", &shift);
    caesar(plain, enc,  shift);
    caesar(enc,   dec, -shift);
    printf("Original  : %s\n", plain);
    printf("Encrypted : %s\n", enc);
    printf("Decrypted : %s\n", dec);
    print_hex(enc);
    return 0;
}
